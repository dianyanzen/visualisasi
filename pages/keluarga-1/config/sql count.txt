//FORMAT TANGGAL
$nowdate = date("d/m/Y");

//COUNT BIODATA
SELECT COUNT(NIK) AS COUNT FROM BIODATA_WNI WHERE TGL_ENTRI >= TO_DATE('".$nowdate."','dd/MM/yyyy') AND TGL_ENTRI < TO_DATE('".$nowdate."','dd/MM/yyyy') +1;
SELECT COUNT(NIK) AS COUNT, TRUNC(TGL_ENTRI) AS TANGGAL FROM BIODATA_WNI  WHERE TGL_ENTRI >= TO_DATE('".$nowdate."','dd/MM/yyyy')-30 AND TGL_ENTRI < TO_DATE('".$nowdate."','dd/MM/yyyy') +1 group by TRUNC(TGL_ENTRI) ORDER BY TRUNC(TGL_ENTRI)

//LAKI-LAKI
SELECT SETUP_KEC.NAMA_KEC, COUNT(NIK) AS LAKI_LAKI FROM DATA_KELUARGA INNER JOIN BIODATA_WNI INNER JOIN SETUP_KEC ON SETUP_KEC.NO_KEC = BIODATA_WNI.NO_KEC ON DATA_KELUARGA.NIK_KK = BIODATA_WNI.NIK  where BIODATA_WNI.JENIS_KLMIN='1' AND TGL_INSERTION < TO_DATE('11/2016','MM/yyyy') +1 AND SETUP_KEC.NO_PROP='32' AND SETUP_KEC.NO_KAB='73' GROUP BY NAMA_KEC ORDER BY NAMA_KEC;
//PEREMPUAN
SELECT SETUP_KEC.NAMA_KEC, COUNT(NIK) AS PEREMPUAN FROM DATA_KELUARGA INNER JOIN BIODATA_WNI INNER JOIN SETUP_KEC ON SETUP_KEC.NO_KEC = BIODATA_WNI.NO_KEC ON DATA_KELUARGA.NIK_KK = BIODATA_WNI.NIK  where BIODATA_WNI.JENIS_KLMIN='2' AND TGL_INSERTION < TO_DATE('11/2016','MM/yyyy') +1 AND SETUP_KEC.NO_PROP='32' AND SETUP_KEC.NO_KAB='73' GROUP BY NAMA_KEC ORDER BY NAMA_KEC;
